-- Dans vos notifs ou locales :

~ r ~ = Rouge
~ b ~ = Bleu
~ g ~ = Vert
~ y ~ = jaune
~ p ~ = Violet
~ c ~ = Gris
~ m ~ = gris foncé
~ u ~ = Noir
~ o ~ = Orange

-- Spécial :

~ n ~ = Nouvelle ligne
~ s ~ = Réinitialiser la couleur
~ h ~ = texte en gras

-- Pour notification :

∑ = Icône Rockstar
÷ = Icône Rockstar
¦ = Icône vérifiée Rockstar

-- Dans les titres de vos serveurs ( onesync obligatoire ) :

^ 1 = rouge
^ 2 = vert
^ 3 = jaune
^ 4 = bleu
^ 5 = CYAN
^ 6 = rose
^ 7 = blanc
^ 8 = bleu/vert foncé
^ 9 = gris
^ 0 = noir
^ 0 est basé sur le RVB du message envoyé
